import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import scipy.stats as stats
from sklearn.preprocessing import StandardScaler, MinMaxScaler

# Cargar los datos (ajusta el nombre del archivo si es diferente)
df = pd.read_csv("SSNMX_catalogo_20250101_20250125.csv", skiprows=4, skipfooter=7, engine="python")

# Filtramos para tener profundidad y magnitud
df = df[["Profundidad", "Magnitud"]].dropna()  # Eliminar valores nulos

# boxplot antes de transformar
plt.figure(figsize=(12, 5))
sns.boxplot(data=df)
plt.title("Boxplot de Profundidad y Magnitud (Antes de Eliminar Outliers)")
plt.show()

#  Z-score
df["Z_Profundidad"] = stats.zscore(df["Profundidad"])
df["Z_Magnitud"] = stats.zscore(df["Magnitud"])

# Definir umbral para outliers (|Z| > 3)
df_no_outliers = df[(abs(df["Z_Profundidad"]) <= 3) & (abs(df["Z_Magnitud"]) <= 3)]

# Eliminar las columnas de Z-score despus de filtrar
df_no_outliers = df_no_outliers[["Profundidad", "Magnitud"]]

#  boxplot despus de eliminar outliers
plt.figure(figsize=(12, 5))
sns.boxplot(data=df_no_outliers)
plt.title("Boxplot de Profundidad y Magnitud (Despus de Eliminar Outliers)")
plt.show()

# Escalamiento Min-Max y Estandarizacin (Z-score)
scaler_minmax = MinMaxScaler()
scaler_zscore = StandardScaler()

df_minmax = pd.DataFrame(scaler_minmax.fit_transform(df_no_outliers), columns=df_no_outliers.columns)
df_zscore = pd.DataFrame(scaler_zscore.fit_transform(df_no_outliers), columns=df_no_outliers.columns)

# Calcular la matriz de correlacin
corr_minmax = df_minmax.corr()
corr_zscore = df_zscore.corr()

# Visualizar la matriz de correlacin
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

sns.heatmap(corr_minmax, annot=True, cmap="coolwarm", ax=axes[0])
axes[0].set_title("Matriz de Correlacin (Min-Max)")

sns.heatmap(corr_zscore, annot=True, cmap="coolwarm", ax=axes[1])
axes[1].set_title("Matriz de Correlacin (Z-score)")

plt.show()
